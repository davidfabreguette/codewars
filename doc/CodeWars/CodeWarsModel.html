def data_store_name
  # We remove module prefix from the equation
  self.class.to_s.underscore.gsub(%r{^code_wars/}, '').pluralize.to_sym
end

def load_attributes(data = {})
  data.each do |key, value|
    send("#{key}=", value) if respond_to? "#{key}="
  end
end
