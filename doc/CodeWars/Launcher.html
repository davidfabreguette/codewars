def launch(event)

  player_name = CodeWars::Player.instance.name || ""
  player_input = nil

  # Shows event label
  ap event.label

  # Search for any decisions choices to provide
  if (decisions = event.decisions).count > 0
    decision_player_attribute = nil

    decisions.each_with_index do |decision, i|
      if decision.current_player_input_attribute.size > 0
        decision_player_attribute = decision.current_player_input_attribute
        break
      end
      ap "#{i+1}/ #{decision.label}"
    end

    player_input = Readline.readline("#{player_name}❞ ", true)

    if decision_player_attribute
      CodeWars::Player.instance.update({
        decision_player_attribute => player_input
        })
    else

    end


  end

  if next_event = resolve_next_event(event, player_input)
    launch(next_event)
  else
    ap "This the end !"
  end

  # current_player = Readline.readline('❞ ', true)
end

  welcome

  # Get the first event
  event = CodeWars::DataStore.instance.events.first
  launch(event)
end
